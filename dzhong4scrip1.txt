CREATE  TABLE Product (ProductID VARCHAR(2) NOT NULL, Description VARCHAR(20), Cost FLOAT, Quantity INT, PRIMARY KEY (ProductID));
CREATE  TABLE Agent (AgentID VARCHAR(2) NOT NULL, FirstName VARCHAR(30), LastName VARCHAR(30), WorkingCity VARCHAR(15), CommissionPercentage FLOAT, PRIMARY KEY (AgentID));
CREATE  TABLE Customer (CustomerID VARCHAR(2) NOT NULL, FirstName VARCHAR(30), LastName VARCHAR(30), City VARCHAR(15), PhoneNumber VARCHAR(12), DealwithAgentID VARCHAR(2), PRIMARY KEY (CustomerID), foreign key (DealwithAgentID) REFERENCES Agent(AgentID));
CREATE  TABLE Puchased (ProductID VARCHAR(2) NOT NULL, CustomerID VARCHAR(2) NOT NULL, Quantity INT, PRIMARY KEY (ProductID, CustomerID), foreign key (ProductID) REFERENCES Product(ProductID), foreign key (CustomerID) REFERENCES Customer(CustomerID));
